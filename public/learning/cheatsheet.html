1. Fade In Animation
@keyframes fade-in { from { opacity: 0; } to { opacity: 1; } } .fade-box { animation: fade-in 2s ease-out; }
2. Slide In From Left
@keyframes slide-in-left { from { transform: translateX(-200px); opacity: 0; } to { transform: translateX(0); opacity: 1; } } .slide-box { animation: slide-in-left 1s ease-out; }
3. Bounce Animation (Infinite Loop)
@keyframes bounce { 0%, 20%, 53%, 80%, 100% { transform: translateY(0); } 40%, 43% { transform: translateY(-30px); } 70% { transform: translateY(-15px); } 90% { transform: translateY(-4px); } } .bounce-box { animation: bounce 2s ease-in-out infinite; }
4. Spin Animation
@keyframes spin { from { transform: rotate(0deg); } to { transform: rotate(360deg); } } .spin-box { animation: spin 2s linear infinite; }
5. Pulse Animation (Scale)
@keyframes pulse { 0% { transform: scale(1); } 50% { transform: scale(1.2); } 100% { transform: scale(1); } } .pulse-box { animation: pulse 1.5s ease-in-out infinite; }
6. Shake Animation
@keyframes shake { 0%, 100% { transform: translateX(0); } 10%, 30%, 50%, 70%, 90% { transform: translateX(-10px); } 20%, 40%, 60%, 80% { transform: translateX(10px); } } .shake-box { animation: shake 0.8s ease-in-out infinite; }
7. Hover Effect (Hover over the box!)
@keyframes hover-grow { from { transform: scale(1); } to { transform: scale(1.1); } } .hover-box:hover { animation: hover-grow 0.3s ease-out forwards; }
8. Complex Multi-Property Animation
@keyframes complex-move { 0% { transform: translateX(0) rotate(0deg) scale(1); background-color: #fd79a8; border-radius: 10px; } 25% { transform: translateX(100px) rotate(90deg) scale(1.2); background-color: #fdcb6e; border-radius: 50%; } /* More keyframes... */ }
9. Delayed Animations (Staggered Effect)
.delayed-box:nth-child(1) { animation-delay: 0.1s; } .delayed-box:nth-child(2) { animation-delay: 0.2s; } .delayed-box:nth-child(3) { animation-delay: 0.3s; } /* Each box starts slightly later */
